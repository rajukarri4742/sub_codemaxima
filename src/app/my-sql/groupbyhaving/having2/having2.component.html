HAVING Statement


QUESION : Display NAME and COUNT of EMPLOYEES GROUPBY SALARY,NAME  Where salary > 20000 field in EMPLOYEE table.

Syntax :    SELECT column1, function_name(column2)
            FROM table_name
            WHERE condition
            GROUP BY column1, column2
            HAVING condition;

                    
        SELECT NAME,SALARY, SUM(SALARY) FROM Employee GROUP BY NAME,SALARY HAVING SUM(SALARY)>20000;
