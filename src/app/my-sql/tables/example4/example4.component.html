CREATE TABLE


QUESION : Create a table named jobs including columns job_id, job_title, min_salary, max_salary and check whether the max_salary amount exceeding the upper limit 25000.

Syntax : CREATE TABLE table_name(
            column1 datatype,
            column2 datatype,
            column3 datatype,
            .....
            columnN datatype
         );

                    
CREATE TABLE IF NOT EXISTS jobs ( 
    JOB_ID varchar(10) NOT NULL , 
    JOB_TITLE varchar(35) NOT NULL, 
    MIN_SALARY decimal(6,0), 
    MAX_SALARY decimal(6,0) 
    CHECK(MAX_SALARY<=25000)
    );
